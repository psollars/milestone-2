def labe_encoder(df):
    label_encoder = LabelEncoder()
    # Separate numeric and object columns
    object_columns = df.select_dtypes(exclude=[np.number]).columns.tolist()
    for c in object_columns:
        df[c] = label_encoder.fit_transform(df[c])
    return df
